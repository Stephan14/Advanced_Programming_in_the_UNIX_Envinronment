代码：
```
#include <stdio.h>  
#include <unistd.h>  
#include <fcntl.h>  
  
  int main(void){  
          if(truncate("a.txt",5)<0){  
                          perror("truncate");  
                          return -1;  
                  }  
    
          int fd = -1;  
          if((fd = open("b.txt",O_RDWR))<0){  
                          perror("open");  
                          return -1;  
                  }  
    
          if(ftruncate(fd,8)<0){  
                          perror("ftruncate");  
                          return -1;  
                  }  
    
          if(fd != -1){  
                          close(fd);  
                  }  
          return 0;  
  } 

```
